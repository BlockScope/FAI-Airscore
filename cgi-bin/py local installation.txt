
###
#Install Python3 with no root privileges

$ cd ~
$ mkdir tmp
$ cd tmp
$ wget https://www.python.org/ftp/python/3.6.2/Python-3.6.2.tgz
$ tar zxvf Python-3.6.2.tgz
$ cd Python-3.6.2
$ ./configure --prefix=$HOME/opt/python-3.6.
$ make
$ make install
$ export PATH=$HOME/opt/python-3.6.2/bin:$PATH
$ . ~/.bash_profile
$ python3 --version

$ nano .bash_profile

SELECT 
	u.useName, 
	u.useEmail 
FROM 
	tblUser u 
	JOIN tblCompPilot b ON u.usePk = b.pilPk 
	LEFT OUTER JOIN (
		SELECT 
			comPk, 
			tasPk, 
			c.traPk, 
			pilPk 
		FROM 
			tblComTaskTrack c 
			JOIN tblTrack t ON c.traPk = t.traPk 
		WHERE 
			tasPk = 11
	) AS a ON b.pilPk = a.pilPK 
	AND b.comPk = a.comPk
